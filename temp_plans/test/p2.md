**Sample Test Code:**

```php
<?php
// tests/Unit/Middleware/ApiAuthenticationTest.php

namespace Tests\Unit\Middleware;

use Tests\TestCase;
use App\Http\Middleware\ApiAuthentication;
use Illuminate\Http\Request;

class ApiAuthenticationTest extends TestCase
{
    private ApiAuthentication $middleware;
    private string $apiKey = 'test-api-key';
    private string $apiSecret = 'test-api-secret';

    protected function setUp(): void
    {
        parent::setUp();
        
        config(['services.asp.api_key' => $this->apiKey]);
        config(['services.asp.api_secret' => $this->apiSecret]);
        
        $this->middleware = new ApiAuthentication();
    }

    public function test_request_without_api_key_returns_401(): void
    {
        $request = Request::create('/api/v1/students', 'GET');
        
        $response = $this->middleware->handle($request, fn($r) => response('OK'));
        
        $this->assertEquals(401, $response->getStatusCode());
    }

    public function test_request_with_valid_credentials_passes(): void
    {
        $timestamp = time();
        $payload = json_encode(['status' => 'pending']);
        $signature = hash_hmac('sha256', $payload . $timestamp, $this->apiSecret);

        $request = Request::create('/api/v1/students', 'GET');
        $request->headers->set('X-API-Key', $this->apiKey);
        $request->headers->set('X-Timestamp', (string) $timestamp);
        $request->headers->set('X-Signature', $signature);

        $response = $this->middleware->handle($request, fn($r) => response('OK'));

        $this->assertEquals(200, $response->getStatusCode());
    }

    public function test_expired_timestamp_returns_401(): void
    {
        $expiredTimestamp = time() - 400; // 6+ minutes ago
        
        $request = Request::create('/api/v1/students', 'GET');
        $request->headers->set('X-API-Key', $this->apiKey);
        $request->headers->set('X-Timestamp', (string) $expiredTimestamp);
        $request->headers->set('X-Signature', 'any-signature');

        $response = $this->middleware->handle($request, fn($r) => response('OK'));

        $this->assertEquals(401, $response->getStatusCode());
    }
}
```

**Sample Integration Test Code:**

```php
<?php
// tests/Feature/Integration/StudentSyncTest.php

namespace Tests\Feature\Integration;

use Tests\TestCase;
use App\Models\Application;
use App\Models\Student;
use Illuminate\Foundation\Testing\RefreshDatabase;

class StudentSyncTest extends TestCase
{
    use RefreshDatabase;

    private string $apiKey;
    private string $apiSecret;

    protected function setUp(): void
    {
        parent::setUp();
        
        $this->apiKey = config('services.asp.api_key');
        $this->apiSecret = config('services.asp.api_secret');
    }

    private function getAuthHeaders(array $payload = []): array
    {
        $timestamp = time();
        $payloadString = json_encode($payload);
        $signature = hash_hmac('sha256', $payloadString . $timestamp, $this->apiSecret);

        return [
            'X-API-Key' => $this->apiKey,
            'X-Timestamp' => $timestamp,
            'X-Signature' => $signature,
        ];
    }

    public function test_get_pending_students_list(): void
    {
        // Arrange
        Student::factory()
            ->has(Application::factory()->state(['status' => 'pending']))
            ->count(5)
            ->create();

        // Act
        $response = $this->getJson('/api/v1/students?status=pending', 
            $this->getAuthHeaders(['status' => 'pending'])
        );

        // Assert
        $response->assertStatus(200)
            ->assertJsonStructure([
                'success',
                'data' => [
                    '*' => ['id', 'student_code', 'full_name', 'status', 'documents']
                ],
                'meta' => ['total', 'per_page', 'current_page']
            ])
            ->assertJsonCount(5, 'data');
    }

    public function test_update_status_to_approved(): void
    {
        // Arrange
        $student = Student::factory()
            ->has(Application::factory()->state(['status' => 'pending']))
            ->create();

        $payload = [
            'application_id' => $student->applications->first()->id,
            'status' => 'approved',
            'student_code' => 'STU2024001',
            'notes' => 'All documents verified'
        ];

        // Act
        $response = $this->postJson('/api/v1/update-status', 
            $payload,
            $this->getAuthHeaders($payload)
        );

        // Assert
        $response->assertStatus(200);
        
        $this->assertDatabaseHas('applications', [
            'id' => $student->applications->first()->id,
            'status' => 'approved'
        ]);
        
        $this->assertDatabaseHas('students', [
            'id' => $student->id,
            'student_code' => 'STU2024001'
        ]);
    }

    public function test_unauthorized_without_api_key(): void
    {
        $response = $this->getJson('/api/v1/students?status=pending');
        
        $response->assertStatus(401);
    }
}
```
